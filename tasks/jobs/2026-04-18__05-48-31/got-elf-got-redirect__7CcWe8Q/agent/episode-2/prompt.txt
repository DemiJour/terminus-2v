func readAtFull(r io.ReaderAt, off int64, b []byte) error {
        n, err := r.ReadAt(b, off)

func readStructAt[T any](r io.ReaderAt, off int64, order binary.ByteOrder, out *T) error {

        }
        return binary.Read(bytes.NewReader(buf), order, out)
}
