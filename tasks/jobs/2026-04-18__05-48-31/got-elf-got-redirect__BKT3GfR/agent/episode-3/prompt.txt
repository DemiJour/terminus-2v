func cStringFrom(b []byte, off uint32) string {
        i := int(off)
        if i < 0 || i >= len(b) {
                return ""
        }
        j := bytes.IndexByte(b[i:], 0)
        if j < 0 {
                return ""
        }
        return string(b[i : i+j])
}

func readAt(b []byte, off uint64, out any) error {
        if off > uint64(len(b)) {
                return io.ErrUnexpectedEOF
        }
        sz := binary.Size(out)
        if sz < 0 {
                return fmt.Errorf("unsupported binary.Size for %T", out)
        }
        end := off + uint64(sz)
        if end > uint64(len(b)) {
                return io.ErrUnexpectedEOF
        }
        return binary.Read(bytes.NewReader(b[off:end]), binary.LittleEndian, out)
}

func writeUint64At(b []byte, off uint64, v uint64) error {
        if off+8 > uint64(len(b)) {
                return io.ErrUnexpectedEOF
        }
        binary.LittleEndian.PutUint64(b[off:off+8], v)
        return nil
}
